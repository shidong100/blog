###MongoDB分片配置###

##mongo sharding环境配置

####config服务器配置文件 config.conf

####shard1服务器配置文件 shard1.conf

####shard2服务器配置文件 shard2.conf

#####shard3服务器配置文件 shard3.conf

####mongos服务器配置文件 mongos.conf

####创建启动停止shell文件
vim ./start-all.sh

启动config server

####编辑停止所有sh文件
vim ./stop-all.sh

启动mongos server

vim ./start-config.sh

启动config server

vim ./stop-config.sh

停止config server

vim ./start-shard1.sh

启动share server

vim ./stop-shard2.sh

停止share server

mongod –shutdown –config /usr/local/mongodb3.4/conf/shard2.conf

vim ./start-shard2.sh

启动share server

mongod –config /usr/local/mongodb3.4/conf/shard2.conf

vim ./stop-shard2.sh

停止share server

mongod –shutdown –config /usr/local/mongodb3.4/conf/shard2.conf

vim ./start-shard3.sh

启动share server

mongod –config /usr/local/mongodb3.4/conf/shard3.conf

vim ./stop-shard3.sh

停止share server

mongod –shutdown –config /usr/local/mongodb3.4/conf/shard3.conf

vim ./start-mongos.sh

启动mongos server

mongos –config /usr/local/mongodb3.4/conf/mongos.conf

vim ./stop-mongos.sh
kill all mongos
